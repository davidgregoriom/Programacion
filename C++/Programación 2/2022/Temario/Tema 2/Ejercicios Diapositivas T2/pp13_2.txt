#include <iostream>
#include <vector>

using namespace std;

class Persona{

public:
  //constructor
  Persona(const string& _nombre, unsigned short int _edad, unsigned short int _altura)
  {
    nombre = _nombre;
    edad = _edad;
    altura = _altura;
  }

  unsigned short edad;
  unsigned short altura;
};
