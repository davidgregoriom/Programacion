#include <iostream>
#include <vector>
using namespace std;

class Persona{

public:
  //constructor
  Persona(const string& _nombre, int _edad,  int _altura)
  {
    setNombre(_nombre);
    setEdad(_edad);
    setAltura(_altura);
  }
  string getNombre(){
     return nombre;
  }
  int getEdad(){
     return edad;
  }
  int getAltura(){
     return altura;
  }
  void setNombre(string _nombre){
          nombre = _nombre;
  }
  void setEdad(int _edad){
      if(_edad>120)
      {//Error
            std::cout<<"Error"<<std::endl;
            edad=0;
      }
      else
      {
          edad = _edad;
      }
  }
  void setAltura(int _altura){
      if(_altura<0)
      {
          std::cout<<"Error"<<std::endl;
          altura=0;
      }
      else
      {
          altura = _altura;
      }
  }


  //variables miembro (atributos)
  string nombre;
  int edad;
  int altura;
};
