#include <iostream>
#include <string>

using namespace std;

class CuentaBanco
{
public:
    std::string NombreCliente;
    float saldo;
    double hipoteca;
};


int main()
{
    CuentaBanco MiCuenta, MiOtraCuenta{"Jose2",500,0};

    MiCuenta.NombreCliente="Jose";
    MiCuenta.saldo=5;
    MiCuenta.hipoteca=-100000;
    
    cout<<"Nombre: "<<MiCuenta.NombreCliente<<", saldo pendiente: "<<MiCuenta.saldo+MiCuenta.hipoteca<<endl;
    cout<<"Nombre: "<<MiOtraCuenta.NombreCliente<<", saldo pendiente: "<<MiOtraCuenta.saldo+MiOtraCuenta.hipoteca<<endl;

    return 0;
}
